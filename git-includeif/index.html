    Git config's includeIf for multiple configs

    tl;dr

    [includeIf "gitdir:~/my/personal/directory"]
  path = ~/.gitconfig.local
  
[includeIf "gitdir:~/my/work/directory"]
  path = ~/.gitconfig.work
  
# In ~/.gitconfig.local
[author]
  email = "personal-email@provider.com"
  name = "first last"
  
# In ~/.gitconfig.work
[author]
  email = "address@work.com
  name = "work-username"
